629 __m128i mul1 = _mm_mul_epu32(upper32_a, b);
630 __m128i mul2 = _mm_mul_epu32(upper32_b, a);

632 __m128i mul3 = _mm_mul_epu32(a, b);
633
634 __m128i high = _mm_slli_epi64(_mm_add_epi64(mul1, mul2), 32);
635 return _mm_add_epi64(high, mul3);
636}

1114template <typename Packet, typename Op>
1115EIGEN_STRONG_INLINE Packet pminmax_propagate_numbers(const Packet& a, const Packet& b, Op op) {
1116 // In this implementation, we take advantage of the fact that pmin/pmax for SSE
1117 // always return a if either a or b is NaN.

1765// We specialize pldexp here, since the generic implementation uses Packet2l, which is not well
1766// supported by SSE, and has more range than is needed for exponents.
1767// TODO(rmlarsen): Remove this specialization once Packet2l has support or casting.
1768template <>
1769EIGEN_STRONG_INLINE Packet2d pldexp<Packet2d>(const Packet2d& a, const Packet2d& exponent) {
1770 // Clamp exponent to [-2099, 2099]
1771 const Packet2d max_exponent = pset1<Packet2d>(2099.0);
1772 const Packet2d e = pmin(pmax(exponent, pnegate(max_exponent)), max_exponent);
1773
1774 // Convert e to integer and swizzle to low-order bits.
1775 const Packet4i ei = vec4i_swizzle1(_mm_cvtpd_epi32(e), 0, 3, 1, 3);
1776
1777 // Split 2^e into four factors and multiply:
1778 const Packet4i bias = _mm_set_epi32(0, 1023, 0, 1023);
1779 Packet4i b = parithmetic_shift_right<2>(ei); // floor(e/4)
1780 Packet2d c = _mm_castsi128_pd(_mm_slli_epi64(padd(b, bias), 52)); // 2^b
1781 Packet2d out = pmul(pmul(pmul(a, c), c), c); // a * 2^(3b)
1782 b = psub(psub(psub(ei, b), b), b); // e - 3b
1783 c = _mm_castsi128_pd(_mm_slli_epi64(padd(b, bias), 52)); // 2^(e - 3b)
1784 out = pmul(out, c); // a * 2^e
1785 return out;
1786}
