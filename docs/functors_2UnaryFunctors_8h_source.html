139template <typename Scalar>
140struct functor_traits<scalar_conjugate_op<Scalar>> {
141 enum {
142 Cost = 0,
143 // Yes the cost is zero even for complexes because in most cases for which
144 // the cost is used, conjugation turns to be a no-op. Some examples:
145 // cost(a*conj(b)) == cost(a*b)
146 // cost(a+conj(b)) == cost(a+b)
147 // <etc.
148 // If we don't set it to zero, then:
149 // A.conjugate().lazyProduct(B.conjugate())
150 // will bake its operands. We definitely don't want that!
151 PacketAccess = packet_traits<Scalar>::HasConj
152 };
153};

355template <typename Scalar>
356struct functor_traits<scalar_exp_op<Scalar>> {
357 enum {
358 PacketAccess = packet_traits<Scalar>::HasExp,
359 // The following numbers are based on the AVX implementation.
360#ifdef EIGEN_VECTORIZE_FMA
361 // Haswell can issue 2 add/mul/madd per cycle.
362 Cost = (sizeof(Scalar) == 4
363 // float: 8 pmadd, 4 pmul, 2 padd/psub, 6 other
364 ? (8 * NumTraits<Scalar>::AddCost + 6 * NumTraits<Scalar>::MulCost)
365 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
366 : (14 * NumTraits<Scalar>::AddCost + 6 * NumTraits<Scalar>::MulCost +
367 scalar_div_cost<Scalar, packet_traits<Scalar>::HasDiv>::value))
368#else
369 Cost = (sizeof(Scalar) == 4
370 // float: 7 pmadd, 6 pmul, 4 padd/psub, 10 other
371 ? (21 * NumTraits<Scalar>::AddCost + 13 * NumTraits<Scalar>::MulCost)
372 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
373 : (23 * NumTraits<Scalar>::AddCost + 12 * NumTraits<Scalar>::MulCost +
374 scalar_div_cost<Scalar, packet_traits<Scalar>::HasDiv>::value))
375#endif
376 };
377};

1220template <>
1221struct scalar_logistic_op<float> {
1222 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE float operator()(const float& x) const {
1223 // Truncate at the first point where the interpolant is exactly one.
1224 const float cst_exp_hi = 16.6355324f;
1225 const float e = numext::exp(numext::mini(x, cst_exp_hi));
1226 return e / (1.0f + e);
1227 }
1228
1229 template <typename Packet>
1230 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Packet packetOp(const Packet& _x) const {
1231 const Packet cst_zero = pset1<Packet>(0.0f);
1232 const Packet cst_one = pset1<Packet>(1.0f);
1233 const Packet cst_half = pset1<Packet>(0.5f);
1234 // Truncate at the first point where the interpolant is exactly one.
1235 const Packet cst_exp_hi = pset1<Packet>(16.6355324f);
1236 const Packet cst_exp_lo = pset1<Packet>(-104.f);
1237
1238 // Clamp x to the non-trivial range where S(x). Outside this
1239 // interval the correctly rounded value of S(x) is either zero
1240 // or one.
1241 Packet zero_mask = pcmp_lt(_x, cst_exp_lo);
1242 Packet x = pmin(_x, cst_exp_hi);
1243
1244 // 1. Multiplicative range reduction:
1245 // Reduce the range of x by a factor of 2. This avoids having
1246 // to compute exp(x) accurately where the result is a denormalized
1247 // value.
1248 x = pmul(x, cst_half);
1249
1250 // 2. Subtractive range reduction:
1251 // Express exp(x) as exp(m*ln(2) + r) = 2^m*exp(r), start by extracting
1252 // m = floor(x/ln(2) + 0.5), such that x = m*ln(2) + r.
1253 const Packet cst_cephes_LOG2EF = pset1<Packet>(1.44269504088896341f);
1254 Packet m = pfloor(pmadd(x, cst_cephes_LOG2EF, cst_half));
1255 // Get r = x - m*ln(2). We use a trick from Cephes where the term
1256 // m*ln(2) is subtracted out in two parts, m*C1+m*C2 = m*ln(2),
1257 // to avoid accumulating truncation errors.
1258 const Packet cst_cephes_exp_C1 = pset1<Packet>(-0.693359375f);
1259 const Packet cst_cephes_exp_C2 = pset1<Packet>(2.12194440e-4f);
1260 Packet r = pmadd(m, cst_cephes_exp_C1, x);
1261 r = pmadd(m, cst_cephes_exp_C2, r);
1262
1263 // 3. Compute an approximation to exp(r) using a degree 5 minimax polynomial.
1264 // We compute even and odd terms separately to increase instruction level
1265 // parallelism.
1266 Packet r2 = pmul(r, r);
1267 const Packet cst_p2 = pset1<Packet>(0.49999141693115234375f);
1268 const Packet cst_p3 = pset1<Packet>(0.16666877269744873046875f);
1269 const Packet cst_p4 = pset1<Packet>(4.1898667812347412109375e-2f);
1270 const Packet cst_p5 = pset1<Packet>(8.33471305668354034423828125e-3f);
1271
1272 const Packet p_even = pmadd(r2, cst_p4, cst_p2);
1273 const Packet p_odd = pmadd(r2, cst_p5, cst_p3);
1274 const Packet p_low = padd(r, cst_one);
1275 Packet p = pmadd(r, p_odd, p_even);
1276 p = pmadd(r2, p, p_low);
1277
1278 // 4. Undo subtractive range reduction exp(m*ln(2) + r) = 2^m * exp(r).
1279 Packet e = pldexp_fast_impl<Packet>::run(p, m);
1280
1281 // 5. Undo multiplicative range reduction by using exp(r) = exp(r/2)^2.
1282 e = pmul(e, e);
1283
1284 // Return exp(x) / (1 + exp(x))
1285 return pselect(zero_mask, cst_zero, pdiv(e, padd(cst_one, e)));
1286 }
1287};
